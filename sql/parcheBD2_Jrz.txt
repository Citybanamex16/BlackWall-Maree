-- Subir esto a la base de datos para la transaccion de EliminarIngrediente y en el delimitador cambiar de ; a $$

CREATE PROCEDURE EliminarIngrediente(IN p_idInsumo VARCHAR(10) CHARACTER SET utf8mb4 COLLATE utf8mb4_spanish2_ci)
BEGIN
  DECLARE v_error_msg VARCHAR(500);
  
  DECLARE EXIT HANDLER FOR SQLEXCEPTION
  BEGIN
    GET DIAGNOSTICS CONDITION 1 v_error_msg = MESSAGE_TEXT;
    ROLLBACK;
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = v_error_msg;
  END;

  START TRANSACTION;
    DELETE FROM producto_tiene_insumo WHERE ID_Insumo = p_idInsumo;
    DELETE FROM insumo WHERE ID_Insumo = p_idInsumo;
  COMMIT;
END$$
